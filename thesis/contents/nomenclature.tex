\begin{nomenclature}
\label{chap:symb}

\begin{longtable}{rl}
  $D_{\text{geo}}$ & 黎曼测地线距离 (Riemannian Geodesic Distance) \\  
  $C_{\text{expr}}^{\text{VG}}$ & 非对称速度引导传输代价 (Asymmetric Velocity-Guided Cost) \\
  $\Pi^*$ & 最优传输耦合矩阵 (Optimal Transport Coupling Matrix) \\
  $U(a, b)$ & 最优传输边缘概率约束可行域 (Feasible Coupling Set) \\
  $\alpha$ & 格罗莫夫-瓦瑟斯坦结构权重 (Gromov-Wasserstein Structural Weight) \\
  $\beta$ & RNA速度惩罚系数 (Velocity Alignment Penalty Weight) \\
  $\vec{V}_i$ & 细胞 $i$ 在物理空间的生物 RNA 速度向量 (RNA Velocity Vector) \\
  $\vec{d}_{ij}$ & 细胞 $i$ 到细胞 $j$ 的空间位移向量 (Spatial Displacement Vector) \\
  $u_\theta(x, t)$ & 神经网络 DriftMLP 拟合的连续时间漂移场 (Continuous Drift Field) \\
  $v(x_t \mid x_0, x_1)$ & 给定初始和边界状态的条件漂移场 (Conditional Vector Field) \\
  $\mathcal{L}_{\text{FM}}$ & 生成式流匹配回归损失函数 (Flow Matching Loss) \\
  $\mathcal{L}_{\text{phy}}$ & RNA速度一致性物理先验损失函数 (Physics-Prior Loss) \\
  $\lambda_{\text{phy}}$ & 物理先验损失项系数 (Physics Loss Regularization Coefficient) \\
  $\sigma(t)$ & 薛定谔桥布朗运动时变条件方差 (Brownian Bridge Variance) \\
  $\Psi(z)$ & 均值漂移势能函数 (Mean Shift Potential Function) \\
  $F_{\text{pot}}(z)$ & 均值漂移势能对流修正负梯度引导力 (Potential Guidance Force) \\
  $\eta$ & 势能引导力强度系数 (Potential Guidance Strength Scale) \\
  $\tau$ & 真空区密度下溢掩膜阈值 (KDE Underflow Density Threshold) \\
  $\Delta t$ & 四阶龙格-库塔数值积分迭代步长 (Runge-Kutta 4th-order Step Size) \\
\end{longtable}

\end{nomenclature}
